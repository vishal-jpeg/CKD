# Variables
$resourceGroupName = "your-resource-group"
$vaultName = "your-vault-name"

# Get the vault
$vault = Get-AzRecoveryServicesVault -Name $vaultName -ResourceGroupName $resourceGroupName

# Set the vault context
Set-AzRecoveryServicesVaultContext -Vault $vault

# Update the redundancy setting to Locally Redundant
Set-AzRecoveryServicesBackupProperty -BackupStorageRedundancy Local

# Optional: Verify the setting
(Get-AzRecoveryServicesBackupProperty).BackupStorageRedundancy
